64  for ( auto &&s: sockets ) {

66  if (s->selectClients & SC_Read)
67  FD_SET(s->fd(), &rd_set);

69  FD_SET(s->fd(), &wr_set);

72  n = max(n, s->fd() + 1);
73  }
74  }
