59  int sync() override {
60  if (sockfd >= 0) {
61  ssize_t n = ::send(sockfd, obuf, pptr() - obuf, 0);
62 
63  if (n < 0) {
64  return -1;
65  } else if (n == (pptr() - obuf)) {
66  setp(obuf, obuf + buffer_size);
67  } else {
68  for (ssize_t cp = n; cp < (buffer_size - n); ++cp) {
69  obuf[cp - n] = obuf[cp];
70  }
71  setp(obuf, obuf + buffer_size);
72  pbump(static_cast<int>(buffer_size - n));
73  }
74  return 0;
75  }
76  return -1;
77  }

92  this->xsputn(&cc, 1);

104  int_type underflow() override {
105  if (sockfd >= 0) {
106  ssize_t n = ::recv(sockfd, ibuf + pushback_size, buffer_size - pushback_size, 0);
107 
108  if (n < 0) {
109  return traits_type::eof();
110  }
111  this->setg(ibuf, ibuf + pushback_size, ibuf + pushback_size + n);
112  if (n) {
113  return traits_type::to_int_type(*(ibuf + pushback_size));
114  }
115  }
116  return traits_type::eof();

281  void findPeerInfo(int (*bind_connect)(int, const struct sockaddr *, socklen_t),
282  list<int> &ai_family_preference) {
283 
284  // Loop over preferences
285  for (auto pref: ai_family_preference) {
286  // And each discovered connection possibility
287  for (struct addrinfo *peer = peer_info; peer != nullptr; peer = peer->ai_next) {
288  // Apply preference
289  if (pref == AF_UNSPEC || pref == peer->ai_family) {
290 
291  // Create a compatible socket
292  sock_fd = ::socket(peer->ai_family, peer->ai_socktype, peer->ai_protocol);
293 
299  if (bind_connect(sock_fd, peer->ai_addr, peer->ai_addrlen)) {
300  error_str = strerror(errno);
301  ::close(sock_fd);
302  sock_fd = -1;
303  } else {
307  memcpy(&peer_addr, peer->ai_addr, peer->ai_addrlen);
308  peer_len = peer->ai_addrlen;
309  af_type = peer->ai_family;
310 
314  int on{1};
315  status = setsockopt(sock_fd, SOL_SOCKET, SO_REUSEADDR, (void *) &on, sizeof(on));
316 
317  break;
318  }
319  }
320  }
321 
322  if (sock_fd >= 0)
323  break;
324  }
