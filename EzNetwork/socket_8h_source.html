60  int sync() override {
61  if (sockfd >= 0) {
62  ssize_t n = ::send(sockfd, obuf, pptr() - obuf, 0);
63 
64  if (n < 0) {
65  return -1;
66  } else if (n == (pptr() - obuf)) {
67  setp(obuf, obuf + buffer_size);
68  } else {
69  for (ssize_t cp = n; cp < (buffer_size - n); ++cp) {
70  obuf[cp - n] = obuf[cp];
71  }
72  setp(obuf, obuf + buffer_size);
73  pbump(static_cast<int>(buffer_size - n));
74  }
75  return 0;
76  }
77  return -1;
78  }

87  int_type overflow(int_type c) override {
88  if (sync() < 0)
89  return traits_type::eof();
90 
91  if (traits_type::not_eof(c)) {
92  char_type cc = traits_type::to_char_type(c);
93  this->xsputn(&cc, 1);
94  }
95 
96  return c;
97  }

105  int_type underflow() override {
106  if (sockfd >= 0) {
107  ssize_t n = ::recv(sockfd, ibuf + pushback_size, buffer_size - pushback_size, 0);
108 
109  if (n < 0) {
110  return traits_type::eof();
111  }
112  this->setg(ibuf, ibuf + pushback_size, ibuf + pushback_size + n);
113  if (n) {
114  return traits_type::to_int_type(*(ibuf + pushback_size));
115  }
116  }
117  return traits_type::eof();
118  }

230  void findPeerInfo(int (*bind_connect)(int, const struct sockaddr *, socklen_t),
231  list<int> &ai_family_preference) {
232 
233  // Loop over preferences
234  for (auto pref: ai_family_preference) {
235  // And each discovered connection possibility
236  for (struct addrinfo *peer = peer_info; peer != nullptr; peer = peer->ai_next) {
237  // Apply preference
238  if (pref == AF_UNSPEC || pref == peer->ai_family) {
239 
240  // Create a compatible socket
241  sock_fd = ::socket(peer->ai_family, peer->ai_socktype, peer->ai_protocol);
242 
248  if (bind_connect(sock_fd, peer->ai_addr, peer->ai_addrlen)) {
249  error_str = strerror(errno);
250  ::close(sock_fd);
251  sock_fd = -1;
252  } else {
256  memcpy(&peer_addr, peer->ai_addr, peer->ai_addrlen);
257  peer_len = peer->ai_addrlen;
258  af_type = peer->ai_family;
259 
263  int on{1};
264  status = setsockopt(sock_fd, SOL_SOCKET, SO_REUSEADDR, (void *) &on, sizeof(on));
265 
266  break;
267  }
268  }
269  }
270 
271  if (sock_fd >= 0)
272  break;
273  }
