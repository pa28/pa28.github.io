138  void findPeerInfo(int (*bind_connect)(int, const struct sockaddr *, socklen_t),
139  int ai_family_preference) {
140 
145  list<int> pref_list{ AF_UNSPEC };
146 
147  // Prepend the user perference if it is not AF_UNSPC, so we don't iterate unnecessarily
148  if (ai_family_preference != AF_UNSPEC)
149  pref_list.push_front( ai_family_preference );
150 
151  // Loop over preferences
152  for (auto pref: pref_list) {
153  // And each discovered connection possibility
154  for (struct addrinfo *peer = peer_info; peer != nullptr; peer = peer->ai_next) {
155  // Apply preference
156  if (pref == AF_UNSPEC || pref == peer->ai_family) {
157 
158  // Create a compatible socket
159  sock_fd = ::socket(peer->ai_family, peer->ai_socktype, peer->ai_protocol);

166  if (bind_connect(sock_fd, peer->ai_addr, peer->ai_addrlen)) {

420  int sync() override {
421  if (socket.sock_fd >= 0) {
422  ssize_t n = ::send(socket.sock_fd, obuf, pptr() - obuf, 0 );
423 
424  if (n < 0) {
425  return -1;
426  } else if (n == (pptr() - obuf)) {
427  setp(obuf, obuf + buffer_size);
428  } else {
429  for (ssize_t cp = n; cp < (buffer_size - n); ++cp) {
430  obuf[cp - n] = obuf[cp];
431  }
432  setp(obuf, obuf + buffer_size);
433  pbump(static_cast<int>(buffer_size-n));
434  }
435  return 0;
436  }
437  return -1;
438  }
