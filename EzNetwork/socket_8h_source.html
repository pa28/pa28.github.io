69  void findPeerInfo(int (*bind_connect)(int, const struct sockaddr *, socklen_t),
70  int ai_family_preference) {
71 
76  list<int> pref_list{ AF_UNSPEC };
77 
78  // Prepend the user perference if it is not AF_UNSPC, so we don't iterate unnecessarily
79  if (ai_family_preference != AF_UNSPEC)
80  pref_list.push_front( ai_family_preference );
81 
82  // Loop over preferences
83  for (auto pref: pref_list) {
84  // And each discovered connection possibility
85  for (struct addrinfo *peer = peerinfo; peer != nullptr; peer = peer->ai_next) {
86  // Apply preference
87  if (pref == AF_UNSPEC || pref == peer->ai_family) {
88 
89  // Create a compatible socket
90  sockfd = ::socket(peer->ai_family, peer->ai_socktype, peer->ai_protocol);
91 
97  if (bind_connect(sockfd, peer->ai_addr, peer->ai_addrlen)) {
98  error_str = strerror(errno);
99  ::close(sockfd);
100  sockfd = -1;
101  } else {
105  memcpy(&peer_addr, peer->ai_addr, peer->ai_addrlen);
106  peer_len = peer->ai_addrlen;
107  af_type = peer->ai_family;
108 
112  int on{1};
113  status = setsockopt(sockfd, SOL_SOCKET, SO_REUSEADDR, (void*)&on, sizeof(on));
114 
115  break;
116  }
117  }
118  }
119 
120  if (sockfd >= 0)
121  break;
122  }

322  explicit socket_streambuf(Socket &sock) : socket(sock), obuf{}, ibuf{} {
323  this->setp(obuf, obuf+buffer_size);
324  this->setg(ibuf, ibuf+pushback_size, ibuf+pushback_size);

327  protected:
328  Socket &socket;
329  char_type obuf[buffer_size];
330  char_type ibuf[buffer_size+pushback_size];
331 
332  int sync() override {
333  if (socket.sockfd >= 0) {
334  ssize_t n = ::send(socket.sockfd, obuf, pptr() - obuf, 0 );
335 
336  if (n < 0) {
337  return -1;
338  } else if (n == (pptr() - obuf)) {
339  setp(obuf, obuf + buffer_size);
340  } else {
341  for (ssize_t cp = n; cp < (buffer_size - n); ++cp) {
342  obuf[cp - n] = obuf[cp];
343  }
344  setp(obuf, obuf + buffer_size);
345  pbump(static_cast<int>(buffer_size-n));
346  }
347  return 0;
348  }
349  return -1;
350  }
351 
352  int_type overflow( int_type c ) override {
353  if (sync() < 0)
354  return traits_type::eof();
355 
356  if (traits_type::not_eof(c)) {
357  char_type cc = traits_type::to_char_type(c);
358  this->xsputn(&cc, 1);
359  }
360 
361  return traits_type::to_int_type(c);
362  }
363 
364  int_type underflow( ) override {
365  if (socket.sockfd >= 0) {
366  ssize_t n = ::recv(socket.sockfd, ibuf + pushback_size, buffer_size - pushback_size, 0);
367 
368  if (n < 0) {
369  return traits_type::eof();
370  }
371  this->setg(ibuf, ibuf + pushback_size, ibuf + pushback_size + n);
372  if (n) {
373  return traits_type::to_int_type(*(ibuf + pushback_size));
374  }
375  }
376  return traits_type::eof();
377  }
